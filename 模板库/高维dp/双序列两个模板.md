// 双序列 DP 通用骨架。换题只动三处:① 边界  ② 怎么记账(从哪个前驱来)  ③ 怎么合并多条来路
auto seqDP = [&](const auto& A, const auto& B) {
    int n = A.size(), m = B.size();
    vector<vector<long long>> dp(n+1, vector<long long>(m+1));  // 存什么按题改:最优/计数→数值; 可行性→0/1

    // ① 边界:一方为空时的答案
    //    最优/代价类填长度或0; 计数类填1; 可行性类填 能否
    for (int i = 0; i <= n; ++i) dp[i][0] = /*base_i*/;
    for (int j = 0; j <= m; ++j) dp[0][j] = /*base_j*/;

    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= m; ++j) {
            // 配对关系按题决定:不一定是"A[i]==B[j]"。
            if (/* 配对条件: 这一步 A[i-1] 与 B[j-1] 能对上 */)
                // ② 对上时怎么记账:通常各消耗一个 → 从对角 dp[i-1][j-1] 来(±1 或 +方案)
                dp[i][j] = /* 用 dp[i-1][j-1] */;
            else
                // ③ 对不上时:放弃一端 → dp[i-1][j](弃A) 与 dp[i][j-1](弃B) 经【合并方式】合并
                //    合并方式:最优→max/min; 计数→相加; 可行性→ ||
                dp[i][j] = /* 合并 dp[i-1][j] 与 dp[i][j-1] */;
        }
    return dp[n][m];   // 注意:连续/以i结尾类的答案在"全表max",不在右下角
};

// 双序列 DP 通用骨架 —— 滚动数组版, O(n*m) 时间, O(m) 空间
auto seqDP = [&](const auto& A, const auto& B) {
    int n = A.size(), m = B.size();
    vector<long long> prev(m + 1), cur(m + 1);   // prev = 上一行(i-1), cur = 当前行(i)

    // ① 边界:i=0 那一行(A 为空)
    for (int j = 0; j <= m; ++j) prev[j] = /*base_j*/;

    for (int i = 1; i <= n; ++i) {
        cur[0] = /*base_i*/;                      // ① 每行第 0 列(B 为空),别忘了重置
        for (int j = 1; j <= m; ++j) {
            if (/* 配对条件:A[i-1] 与 B[j-1] 能对上 */)
                cur[j] = /* ② 用 prev[j-1] (原 dp[i-1][j-1],对角) */;
            else
                cur[j] = /* ③ 合并 prev[j] (弃A,原dp[i-1][j]) 与 cur[j-1] (弃B,原dp[i][j-1]) */;
        }
        swap(prev, cur);                          // 滚动:本行变下一轮的上一行
    }
    return prev[m];                               // swap 之后答案在 prev,不是 cur!
};
