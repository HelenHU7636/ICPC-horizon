// 网格 DP 通用骨架。换题只动四处:① 存什么  ② 前驱邻居  ③ 怎么合并  ④ 本格条件/贡献
auto gridDP = [&](const vector<vector<int>>& g) {
    int n = g.size(), m = g[0].size();
    vector<vector<long long>> dp(n, vector<long long>(m, 0));

    for (int i = 0; i < n; ++i)
        for (int j = 0; j < m; ++j) {
            if (i == 0 && j == 0) { dp[i][j] = /*① 起点初值*/; continue; }

            // ④ 本格条件:障碍/不可用 → dp[i][j]=0(或跳过),否则往下转移
            if (/* g[i][j] 不可用 */) { dp[i][j] = 0; continue; }

            // ② 取固定邻居;越界的那个填"合并方式的单位元"(见下)
            long long up   = (i > 0) ? dp[i-1][j]   : /*越界默认*/;
            long long left = (j > 0) ? dp[i][j-1]   : /*越界默认*/;
            // long long diag = (i>0&&j>0) ? dp[i-1][j-1] : /*越界默认*/;  // 需要对角时打开

            // ③ 合并 up/left[/diag] + ④ 本格贡献
            dp[i][j] = /* 合并(up,left[,diag]) */ + /* 本格贡献:g[i][j] 或 0 */;
        }
    return dp[n-1][m-1];   // 路径类在右下角;最大正方形/矩形类在【全表 max】
};

| 旋钮 | 路径计数 | 最小路径和 | 带障碍路径 | 最大正方形 |
|---|---|---|---|---|
| ① 存什么 | 到(i,j)的路径数 | 到(i,j)最小和 | 路径数 | 以(i,j)为右下角的最大边长 |
| ① 起点初值 | 1 | g[0][0] | 障碍?0:1 | g[0][0]==1?1:0 |
| ② 邻居 | 上+左 | 上+左 | 上+左 | 上+左+对角 |
| ③ 合并 | 相加 | min | 相加 | min(三者)+1 |
| ④ 本格 | +0 | +g[i][j] | 障碍则=0 | g==1才走,否则0 |
| 答案 | 右下角 | 右下角 | 右下角 | 全表 max |

// O(m) 空间,仅适用"上+左"两邻居(路径计数/最小路径和/带障碍)
vector<long long> dp(m);
for (int j = 0; j < m; ++j) dp[j] = /* 第一行的值 */;   // 初始化第 0 行
for (int i = 1; i < n; ++i) {
    dp[0] = /* 合并(dp[0]=上) + 本格 */;                // 第 0 列只有"上"
    for (int j = 1; j < m; ++j)
        // 关键:此刻 dp[j] 还没被改 = 上一行的值 = "上"
        //       dp[j-1] 本轮已更新 = 本行 = "左"
        dp[j] = /* 合并(dp[j]=上, dp[j-1]=左) + 本格 */;
}
return dp[m-1];
